## how to use

1. import css file in main.tsx or app.tsx file

```tsx
import "../node_modules/@milon27/react-sample/dist/react-sidebar.css";
```

1. use the component

```tsx
import { Sample } from "@milon27/react-sample";
function App() {
  return (
    <>
      <Sample />
    </>
  );
}

export default App;
```
